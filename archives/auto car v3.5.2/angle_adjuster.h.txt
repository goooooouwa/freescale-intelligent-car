#define MID_ANGLE 25

//DECLARATIONS
void angleAdjuster(int steer_angle){
    //直线 |steer_angle| < 15
    if((steer_angle<15)&&(steer_angle>-15)){
        //高速直线
        PWMDTY01=30;
        PWMDTY67 = (152+steer_angle)*5;  
    }
    
    //缓弯 15 <= |steer_angle| < MID_ANGLE
    if((steer_angle<MID_ANGLE && steer_angle>=15)||(steer_angle<=-15 && steer_angle>-MID_ANGLE)){
        //低速过弯
        PWMDTY01=30;
        PWMDTY67 = (152+steer_angle)*5;  
        //PWMDTY67 = (int)((152+1.8*(float)steer_angle)*5);  
    }
    
    //急弯 MID_ANGLE <= |steer_angle| < MAX_ANGLE
    if((steer_angle<MAX_ANGLE && steer_angle>=MID_ANGLE)||(steer_angle<=-MID_ANGLE && steer_angle>-MAX_ANGLE)){
        //急弯转角大，车子很难前进，加大马力避免停滞
        PWMDTY01=40;
        PWMDTY67 = (152+steer_angle)*5;  
    }
    
    //弯道过急 |steer_angle| >= MAX_ANGLE
    if(steer_angle>=60){
        //转角超过车子极限，限制其等于极限值
        PWMDTY01=60;
        PWMDTY67 = (152+MAX_ANGLE)*5;  
    }
    if(steer_angle<=-50){
        PWMDTY01=60;
        PWMDTY67 = (152+(-MAX_ANGLE))*5;  
    }    
     
}
